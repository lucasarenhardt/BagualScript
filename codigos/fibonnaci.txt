intche limite;
MOSTRE TCHE("Insira o numero: ");
DIGA GURI (limite);

intche a = 0;
intche b = 1;
intche temp = 0;

MOSTRE TCHE ("Sequência de Fibonacci até ", limite, ": ");

TROTEANDO (intche i = 0; a <= limite; i++){
    MOSTRE TCHE (a, " ");
    temp = a + b;
    a = b;
    b = temp;
}

MOSTRE TCHE ("\n");
